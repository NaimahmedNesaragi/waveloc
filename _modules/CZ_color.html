    """

    :param dt:
    :type dt: float

    :returns: List of 3 values between 0 and 1 for color indexing.

    """

    tt = range(0, 105, 5)
    liste = [dt-tt[i] for i in range(len(tt))]
    ind1 = np.where(np.abs(liste) == np.min(np.abs(liste)))[0]
    ind2 = ind1[0]
    ind = int(np.max(round(len(cmap)*ind2/len(tt))))
    color = cmap[ind]

    return color
